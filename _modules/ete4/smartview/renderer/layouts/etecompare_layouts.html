    def set_node_style(self, node):
        difference = node.props.get("compare_distance")
        if difference:
            node.sm_style["fgcolor"] = self.diff_node_color
            node.sm_style["size"] = self.diff_node_size
            node.sm_style["fgopacity"] = float(difference)
